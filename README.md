# Book
Demo Express APi for book

# npm init -y 

# npm i express nodemon mongoose

# Endpoints

# Create book - POST - C
http://localhost:3000/books

# Get all books - GET - R
http://localhost:3000/books

# Get single book - GET - R
http://localhost:3000/books/{id}

# update book - PUT  - U
http://localhost:3000/books/{id}

# delete book - DELETE - D
http://localhost:3000/books/{id}
